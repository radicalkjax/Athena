6/// Different kinds of ICMPv6 messages.
7///
8/// The data stored in this enum corresponds to the statically sized data
9/// at the start of an ICMPv6 packet without the checksum. If you also need
10/// the checksum you can package and [`Icmpv6Type`] value in an [`Icmpv6Header`]
11/// struct.
12///
13/// # Decoding Example (complete packet):
14///
15/// ```
16/// # use etherparse::{PacketBuilder};
17/// # let mut builder = PacketBuilder::
18/// #   ethernet2([0;6], [0;6])
19/// #   .ipv6([0;16], [0;16], 20)
20/// #   .icmpv6_echo_request(1, 2);
21/// # let payload = [1,2,3,4];
22/// # let mut packet = Vec::<u8>::with_capacity(builder.size(payload.len()));
23/// # builder.write(&mut packet, &payload);
24/// use etherparse::PacketHeaders;
25///
26/// let headers = PacketHeaders::from_ethernet_slice(&packet).unwrap();
27///
28/// use etherparse::TransportHeader::*;
29/// match headers.transport {
30///     Some(Icmpv6(icmp)) => {
31///         use etherparse::Icmpv6Type::*;
32///         match icmp.icmp_type {
33///             // Unknown is used when further decoding is currently not supported for the icmp type & code.
34///             // You can still further decode the packet on your own by using the raw data in this enum
35///             // together with `headers.payload` (contains the packet data after the 8th byte)
36///             Unknown{ type_u8, code_u8, bytes5to8 } => println!("Unknown{{ type_u8: {}, code_u8: {}, bytes5to8: {:?} }}", type_u8, code_u8, bytes5to8),
37///             DestinationUnreachable(header) => println!("{:?}", header),
38///             PacketTooBig { mtu } => println!("TimeExceeded{{ mtu: {} }}", mtu),
39///             TimeExceeded(code) => println!("{:?}", code),
40///             ParameterProblem(header) => println!("{:?}", header),
41///             EchoRequest(header) => println!("{:?}", header),
42///             EchoReply(header) => println!("{:?}", header),
43///         }
44///     },
45///     _ => {},
46/// }
47/// ```
48///
49/// # Encoding Example (only ICMPv6 part)
50///
51/// To get the on wire bytes of an Icmpv6Type it needs to get packaged
52/// into a [`Icmpv6Header`] so the checksum gets calculated.
53///
54/// ```
55/// # use etherparse::Ipv6Header;
56/// # let ip_header: Ipv6Header = Default::default();
57/// # let invoking_packet : [u8;0] = [];
58///
59/// use etherparse::{Icmpv6Type, icmpv6::DestUnreachableCode};
60/// let t = Icmpv6Type::DestinationUnreachable(
61///     DestUnreachableCode::Address
62/// );
63///
64/// // to calculate the checksum the ip header and the payload
65/// // (in case of dest unreachable the invoking packet) are needed
66/// let header = t.to_header(ip_header.source, ip_header.destination, &invoking_packet).unwrap();
67///
68/// // an ICMPv6 packet is composed of the header and payload
69/// let mut packet = Vec::with_capacity(header.header_len() + invoking_packet.len());
70/// packet.extend_from_slice(&header.to_bytes());
71/// packet.extend_from_slice(&invoking_packet);
72/// #
73/// # {
74/// #   let checksum_be = header.checksum.to_be_bytes();
75/// #   assert_eq!(
76/// #       &packet,
77/// #       &[
78/// #           header.icmp_type.type_u8(),
79/// #           header.icmp_type.code_u8(),
80/// #           checksum_be[0],
81/// #           checksum_be[1],
82/// #           0,0,0,0
83/// #       ]
84/// #   );
85/// # }
86/// ```
87#[derive(Clone, Copy, Debug, PartialEq, Eq)]
88pub enum Icmpv6Type {
89    /// In case of an unknown icmp type is received the header elements of
90    /// the first 8 bytes/octets are stored raw in this enum value.
91    ///
92    /// # What is part of the header for `Icmpv6Type::Unknown`?
93    ///
94    /// For unknown ICMPv6 type & code combination the first 8 bytes are stored
95    /// in the [`Icmpv6Header`] and the rest is stored in the payload
96    /// ([`Icmpv6Slice::payload`] or [`PacketHeaders::payload`]).
97    ///
98    ///
99    /// ```text
100    /// 0               1               2               3               4
101    /// +---------------------------------------------------------------+  -
102    /// |     type_u8   |    code_u8    |  checksum (in Icmpv6Header)   |  |
103    /// +---------------------------------------------------------------+  | part of header & type
104    /// |                          bytes5to8                            |  ↓
105    /// +---------------------------------------------------------------+  -
106    /// |                                                               |  |
107    /// ...                           ...                             ...  | part of payload
108    /// |                                                               |  ↓
109    /// +---------------------------------------------------------------+  -
110    /// ```
111    Unknown {
112        /// ICMPv6 type (present in the first byte of the ICMPv6 packet).
113        type_u8: u8,
114        /// ICMPv6 code (present in the 2nd byte of the ICMPv6 packet).
115        code_u8: u8,
116        /// Bytes located at th 5th, 6th, 7th and 8th position of the ICMP packet.
117        bytes5to8: [u8; 4],
118    },
119
120    /// Message sent to inform the client that the destination is unreachable for
121    /// some reason.
122    ///
123    /// # What is part of the header for `Icmpv6Type::DestinationUnreachable`?
124    ///
125    /// For the `Icmpv6Type::DestinationUnreachable` type the first 8 bytes/octets of the ICMPv6
126    /// packet are part of the header. The `unused` part is not stored and droped.
127    /// The offending packet is stored in the payload part of the packet
128    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and is not part of
129    /// the [`Icmpv6Header`].
130    ///
131    /// ```text
132    /// 0               1               2               3               4
133    /// +---------------------------------------------------------------+  -
134    /// |       1       | [value as u8] |  checksum (in Icmpv6Header)   |  |
135    /// +---------------------------------------------------------------+  | part of header & type
136    /// |                           <unused>                            |  ↓
137    /// +---------------------------------------------------------------+  -
138    /// |                                                               |  |
139    /// |     <As much of invoking packet as possible without           |  | part of payload
140    /// ...   the ICMPv6 packet exceeding the minimum IPv6 MTU>       ...  |
141    /// |                                                               |  ↓
142    /// +---------------------------------------------------------------+  -
143    /// ```
144    ///
145    /// # RFC 4443 Description
146    ///
147    /// A Destination Unreachable message SHOULD be generated by a router, or
148    /// by the IPv6 layer in the originating node, in response to a packet
149    /// that cannot be delivered to its destination address for reasons other
150    /// than congestion.  (An ICMPv6 message MUST NOT be generated if a
151    /// packet is dropped due to congestion.)
152    DestinationUnreachable(icmpv6::DestUnreachableCode),
153
154    /// Sent if a packet to too big to be forwarded.
155    ///
156    /// # What is part of the header for `Icmpv6Type::PacketTooBig`?
157    ///
158    /// For the `Icmpv6Type::PacketTooBig` type the first 8 bytes/octets of the ICMPv6
159    /// packet are part of the header. The offending packet is stored in the payload part of the packet
160    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and is not part of
161    /// the [`Icmpv6Header`].
162    ///
163    /// ```text
164    /// 0               1               2               3               4
165    /// +---------------------------------------------------------------+  -
166    /// |       2       |       0       |  checksum (in Icmpv6Header)   |  |
167    /// +---------------------------------------------------------------+  | part of header & type
168    /// |                              mtu                              |  ↓
169    /// +---------------------------------------------------------------+  -
170    /// |                                                               |  |
171    /// |     <As much of invoking packet as possible without           |  | part of payload
172    /// ...   the ICMPv6 packet exceeding the minimum IPv6 MTU>       ...  |
173    /// |                                                               |  ↓
174    /// +---------------------------------------------------------------+  -
175    /// ```
176    ///
177    /// # RFC 4443 Description
178    ///
179    /// A Packet Too Big MUST be sent by a router in response to a packet
180    /// that it cannot forward because the packet is larger than the MTU of
181    /// the outgoing link.  The information in this message is used as part
182    /// of the Path MTU Discovery process.
183    PacketTooBig {
184        /// The Maximum Transmission Unit of the next-hop link.
185        mtu: u32,
186    },
187
188    /// Generated when a datagram had to be discarded due to the hop limit field
189    /// reaching zero.
190    ///
191    /// # What is part of the header for `Icmpv6Type::TimeExceeded`?
192    ///
193    /// For the `Icmpv6Type::TimeExceeded` type the first 8 bytes/octets of the ICMPv6
194    /// packet are part of the header. The `unused` part is not stored and droped.
195    /// The offending packet is stored in the payload part of the packet
196    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and is not part of
197    /// the [`Icmpv6Header`].
198    ///
199    /// ```text
200    /// 0               1               2               3               4
201    /// +---------------------------------------------------------------+  -
202    /// |       3       | [value as u8] |  checksum (in Icmpv6Header)   |  |
203    /// +---------------------------------------------------------------+  | part of header & type
204    /// |                           <unused>                            |  ↓
205    /// +---------------------------------------------------------------+  -
206    /// |                                                               |  |
207    /// |     <As much of invoking packet as possible without           |  | part of payload
208    /// ...   the ICMPv6 packet exceeding the minimum IPv6 MTU>       ...  |
209    /// |                                                               |  ↓
210    /// +---------------------------------------------------------------+  -
211    /// ```
212    ///
213    /// # RFC 4443 Description
214    ///
215    /// If a router receives a packet with a Hop Limit of zero, or if a
216    /// router decrements a packet's Hop Limit to zero, it MUST discard the
217    /// packet and originate an ICMPv6 Time Exceeded message with Code 0 to
218    /// the source of the packet.  This indicates either a routing loop or
219    /// too small an initial Hop Limit value.
220    ///
221    /// An ICMPv6 Time Exceeded message with Code 1 is used to report
222    /// fragment reassembly timeout, as specified in [IPv6, Section 4.5].
223    TimeExceeded(icmpv6::TimeExceededCode),
224
225    /// Sent if there is a problem with a parameter in a received packet.
226    ///
227    /// # What is part of the header for `Icmpv6Type::ParameterProblem`?
228    ///
229    /// For the `Icmpv6Type::ParameterProblem` type the first 8 bytes/octets of the ICMPv6
230    /// packet are part of the header. The `unused` part is not stored and droped.
231    /// The offending packet is stored in the payload part of the packet
232    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and is not part of
233    /// the [`Icmpv6Header`].
234    ///
235    /// ```text
236    /// 0               1               2               3               4
237    /// +---------------------------------------------------------------+  -
238    /// |       4       | [value].code  |  checksum (in Icmpv6Header)   |  |
239    /// +---------------------------------------------------------------+  | part of header & type
240    /// |                        [value].pointer                        |  ↓
241    /// +---------------------------------------------------------------+  -
242    /// |                                                               |  |
243    /// |     <As much of invoking packet as possible without           |  | part of payload
244    /// ...   the ICMPv6 packet exceeding the minimum IPv6 MTU>       ...  |
245    /// |                                                               |  ↓
246    /// +---------------------------------------------------------------+  -
247    /// ```
248    ///
249    /// # RFC 4443 Description
250    ///
251    /// If an IPv6 node processing a packet finds a problem with a field in
252    /// the IPv6 header or extension headers such that it cannot complete
253    /// processing the packet, it MUST discard the packet and SHOULD
254    /// originate an ICMPv6 Parameter Problem message to the packet's source,
255    /// indicating the type and location of the problem.
256    ParameterProblem(icmpv6::ParameterProblemHeader),
257
258    /// Requesting an `EchoReply` from the receiver.
259    ///
260    /// # What is part of the header for `Icmpv6Type::EchoRequest`?
261    ///
262    /// For the [`Icmpv6Type::EchoRequest`] type the first 8 bytes/octets of the
263    /// ICMPv6 packet are part of the header. This includes the `id` and `seq`
264    /// fields. The data part of the ICMP echo request packet is part of the payload
265    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and not part of the
266    /// [`Icmpv6Header`].
267    ///
268    /// ```text
269    /// 0               1               2               3               4
270    /// +---------------------------------------------------------------+  -
271    /// |      128      |       0       |  checksum (in Icmpv6Header)   |  |
272    /// +---------------------------------------------------------------+  | part of header & type
273    /// |          [value].id           |         [value].seq           |  ↓
274    /// +---------------------------------------------------------------+  -
275    /// |                                                               |  |
276    /// ...                          <data>                           ...  | part of payload
277    /// |                                                               |  ↓
278    /// +---------------------------------------------------------------+  -
279    /// ```
280    ///
281    /// # RFC 4443 Description
282    ///
283    /// Every node MUST implement an ICMPv6 Echo responder function that
284    /// receives Echo Requests and originates corresponding Echo Replies.  A
285    /// node SHOULD also implement an application-layer interface for
286    /// originating Echo Requests and receiving Echo Replies, for diagnostic
287    /// purposes.
288    EchoRequest(IcmpEchoHeader),
289    /// Response to an `EchoRequest` message.
290    ///
291    /// # What is part of the header for `Icmpv6Type::EchoReply`?
292    ///
293    /// For the [`Icmpv6Type::EchoReply`] type the first 8 bytes/octets of the
294    /// ICMPv6 packet are part of the header. This includes the `id` and `seq`
295    /// fields. The data part of the ICMP echo request packet is part of the payload
296    /// ([`Icmpv6Slice::payload`] & [`PacketHeaders::payload`]) and not part of the
297    /// [`Icmpv6Header`].
298    ///
299    /// ```text
300    /// 0               1               2               3               4
301    /// +---------------------------------------------------------------+  -
302    /// |      129      |       0       |  checksum (in Icmpv6Header)   |  |
303    /// +---------------------------------------------------------------+  | part of header & type
304    /// |          [value].id           |         [value].seq           |  ↓
305    /// +---------------------------------------------------------------+  -
306    /// |                                                               |  |
307    /// ...                          <data>                           ...  | part of payload
308    /// |                                                               |  ↓
309    /// +---------------------------------------------------------------+  -
310    /// ```
311    ///
312    /// # RFC 4443 Description
313    ///
314    /// Every node MUST implement an ICMPv6 Echo responder function that
315    /// receives Echo Requests and originates corresponding Echo Replies. A
316    /// node SHOULD also implement an application-layer interface for
317    /// originating Echo Requests and receiving Echo Replies, for diagnostic
318    /// purposes.
319    ///
320    /// The source address of an Echo Reply sent in response to a unicast
321    /// Echo Request message MUST be the same as the destination address of
322    /// that Echo Request message.
323    ///
324    /// An Echo Reply SHOULD be sent in response to an Echo Request message
325    /// sent to an IPv6 multicast or anycast address.  In this case, the
326    /// source address of the reply MUST be a unicast address belonging to
327    /// the interface on which the Echo Request message was received.
328    ///
329    /// The data received in the ICMPv6 Echo Request message MUST be returned
330    /// entirely and unmodified in the ICMPv6 Echo Reply message.
331    EchoReply(IcmpEchoHeader),
332}

334impl Icmpv6Type {
335    /// Returns the type value (first byte of the ICMPv6 header) of this type.
336    #[inline]
337    pub fn type_u8(&self) -> u8 {
338        use crate::{icmpv6::*, Icmpv6Type::*};
339        match self {
340            Unknown {
341                type_u8,
342                code_u8: _,
343                bytes5to8: _,
344            } => *type_u8,
345            DestinationUnreachable(_) => TYPE_DST_UNREACH,
346            PacketTooBig { mtu: _ } => TYPE_PACKET_TOO_BIG,
347            TimeExceeded(_) => TYPE_TIME_EXCEEDED,
348            ParameterProblem(_) => TYPE_PARAMETER_PROBLEM,
349            EchoRequest(_) => TYPE_ECHO_REQUEST,
350            EchoReply(_) => TYPE_ECHO_REPLY,
351        }
352    }
353
354    /// Returns the code value (second byte of the ICMPv6 header) of this type.
355    #[inline]
356    pub fn code_u8(&self) -> u8 {
357        use Icmpv6Type::*;
358        match self {
359            Unknown {
360                type_u8: _,
361                code_u8,
362                bytes5to8: _,
363            } => *code_u8,
364            DestinationUnreachable(code) => code.code_u8(),
365            PacketTooBig { mtu: _ } => 0,
366            TimeExceeded(code) => code.code_u8(),
367            ParameterProblem(header) => header.code.code_u8(),
368            EchoRequest(_) => 0,
369            EchoReply(_) => 0,
370        }
371    }
372
373    /// Calculates the checksum of the ICMPv6 header.
374    ///
375    /// <p style="background:rgba(255,181,77,0.16);padding:0.75em;">
376    /// <strong>Warning:</strong> Don't use this method to verfy if a checksum of a
377    /// received packet is correct. This method assumes that all unused bytes are
378    /// filled with zeros. If this is not the case the computed checksum value will
379    /// will be incorrect for a received packet.
380    ///
381    /// If you want to verify that a received packet has a correct checksum use
382    /// [`Icmpv6Slice::is_checksum_valid`] instead.
383    /// </p>
384    pub fn calc_checksum(
385        &self,
386        source_ip: [u8; 16],
387        destination_ip: [u8; 16],
388        payload: &[u8],
389    ) -> Result<u16, ValueTooBigError<usize>> {
390        // check that the total length fits into the field
391        //
392        // Note according to RFC 2460 the "Upper-Layer Packet Length" used
393        // in the checksum calculation, for protocols that don't contain
394        // their own length information (like ICMPv6), is "the Payload Length
395        // from the IPv6 header, minus the length of any extension headers present
396        // between the IPv6 header and the upper-layer header."
397        let max_payload_len: usize = (core::u32::MAX as usize) - self.header_len();
398        if max_payload_len < payload.len() {
399            return Err(ValueTooBigError {
400                actual: payload.len(),
401                max_allowed: max_payload_len,
402                value_type: ValueType::Icmpv6PayloadLength,
403            });
404        }
405
406        let msg_len = payload.len() + self.header_len();
407
408        // calculate the checksum
409        // NOTE: rfc4443 section 2.3 - Icmp6 *does* use a pseudoheader,
410        // unlike Icmp4
411        let pseudo_sum = checksum::Sum16BitWords::new()
412            .add_16bytes(source_ip)
413            .add_16bytes(destination_ip)
414            .add_2bytes([0, ip_number::IPV6_ICMP.0])
415            .add_4bytes((msg_len as u32).to_be_bytes());
416
417        use crate::{icmpv6::*, Icmpv6Type::*};
418        Ok(match self {
419            Unknown {
420                type_u8,
421                code_u8,
422                bytes5to8,
423            } => pseudo_sum
424                .add_2bytes([*type_u8, *code_u8])
425                .add_4bytes(*bytes5to8),
426            DestinationUnreachable(header) => {
427                pseudo_sum.add_2bytes([TYPE_DST_UNREACH, header.code_u8()])
428            }
429            PacketTooBig { mtu } => pseudo_sum
430                .add_2bytes([TYPE_PACKET_TOO_BIG, 0])
431                .add_4bytes(mtu.to_be_bytes()),
432            TimeExceeded(code) => pseudo_sum.add_2bytes([TYPE_TIME_EXCEEDED, code.code_u8()]),
433            ParameterProblem(header) => pseudo_sum
434                .add_2bytes([TYPE_PARAMETER_PROBLEM, header.code.code_u8()])
435                .add_4bytes(header.pointer.to_be_bytes()),
436            EchoRequest(echo) => pseudo_sum
437                .add_2bytes([TYPE_ECHO_REQUEST, 0])
438                .add_4bytes(echo.to_bytes()),
439            EchoReply(echo) => pseudo_sum
440                .add_2bytes([TYPE_ECHO_REPLY, 0])
441                .add_4bytes(echo.to_bytes()),
442        }
443        .add_slice(payload)
444        .ones_complement()
445        .to_be())
446    }
447
448    /// Creates a header with the correct checksum.
449    pub fn to_header(
450        self,
451        source_ip: [u8; 16],
452        destination_ip: [u8; 16],
453        payload: &[u8],
454    ) -> Result<Icmpv6Header, ValueTooBigError<usize>> {
455        Ok(Icmpv6Header {
456            checksum: self.calc_checksum(source_ip, destination_ip, payload)?,
457            icmp_type: self,
458        })
459    }
460
461    /// Serialized length of the header in bytes/octets.
462    ///
463    /// Note that this size is not the size of the entire
464    /// ICMPv6 packet but only the header.
465    pub fn header_len(&self) -> usize {
466        use Icmpv6Type::*;
467        match self {
468            Unknown {
469                type_u8: _,
470                code_u8: _,
471                bytes5to8: _,
472            }
473            | DestinationUnreachable(_)
474            | PacketTooBig { mtu: _ }
475            | TimeExceeded(_)
476            | ParameterProblem(_)
477            | EchoRequest(_)
478            | EchoReply(_) => 8,
479        }
480    }
481
482    /// If the ICMP type has a fixed size returns the number of
483    /// bytes that should be present after the header of this type.
484    #[inline]
485    pub fn fixed_payload_size(&self) -> Option<usize> {
486        use Icmpv6Type::*;
487        match self {
488            Unknown {
489                type_u8: _,
490                code_u8: _,
491                bytes5to8: _,
492            }
493            | DestinationUnreachable(_)
494            | PacketTooBig { mtu: _ }
495            | TimeExceeded(_)
496            | ParameterProblem(_)
497            | EchoRequest(_)
498            | EchoReply(_) => None,
499        }
500    }

516        #[test]
517        fn type_u8(
518            code_u8 in any::<u8>(),
519            bytes5to8 in any::<[u8;4]>(),
520        ) {
521            {
522                let type_u8_type_pair = [
523                    (TYPE_DST_UNREACH, DestinationUnreachable(DestUnreachableCode::SourceAddressFailedPolicy)),
524                    (TYPE_PACKET_TOO_BIG, PacketTooBig{ mtu: u32::from_be_bytes(bytes5to8), }),
525                    (TYPE_TIME_EXCEEDED, TimeExceeded(TimeExceededCode::HopLimitExceeded)),
526                    (TYPE_PARAMETER_PROBLEM, ParameterProblem(ParameterProblemHeader{ code: ParameterProblemCode::UnrecognizedNextHeader, pointer: u32::from_be_bytes(bytes5to8)})),
527                    (TYPE_ECHO_REQUEST, EchoRequest(IcmpEchoHeader::from_bytes(bytes5to8))),
528                    (TYPE_ECHO_REPLY, EchoReply(IcmpEchoHeader::from_bytes(bytes5to8))),
529                ];
530                for test in type_u8_type_pair {
531                    assert_eq!(test.0, test.1.type_u8());
532                }
533            }
534
535            for t in 0..=u8::MAX {
536                assert_eq!(
537                    t,
538                    Unknown{
539                        type_u8: t,
540                        code_u8,
541                        bytes5to8,
542                    }.type_u8()
543                );
544            }
545        }
546    }

549        #[test]
550        fn code_u8(
551            code_u8 in any::<u8>(),
552            bytes5to8 in any::<[u8;4]>(),
553        ) {
554            // types with 0 as code
555            {
556                let code_type_pair = [
557                    (0, PacketTooBig{ mtu: u32::from_be_bytes(bytes5to8), }),
558                    (0, EchoRequest(IcmpEchoHeader::from_bytes(bytes5to8))),
559                    (0, EchoReply(IcmpEchoHeader::from_bytes(bytes5to8))),
560                ];
561                for test in code_type_pair {

564            }
565
566            // destination unreachable

569            }

571            // time exceeded

573                assert_eq!(code_u8, TimeExceeded(code).code_u8());
574            }
575
576            // parameter problem
577            for (code, code_u8) in parameter_problem_code_test_consts::VALID_VALUES {
578                assert_eq!(
579                    code_u8,
580                    ParameterProblem(
581                        ParameterProblemHeader{
582                            code,
583                            pointer: u32::from_be_bytes(bytes5to8),
584                        }
585                    ).code_u8()
586                );
587            }
588

604        #[test]
605        #[cfg(not(target_pointer_width = "32"))]
606        fn calc_checksum(
607            ip_header in ipv6_any(),
608            icmpv6_type in icmpv6_type_any(),
609            type_u8 in any::<u8>(),
610            code_u8 in any::<u8>(),
611            bytes5to8 in any::<[u8;4]>(),
612            // max length is u32::MAX - header_len (7)
613            bad_len in (core::u32::MAX - 7) as usize..=(core::isize::MAX as usize),
614            payload in proptest::collection::vec(any::<u8>(), 0..64)
615        ) {
616            use Icmpv6Type::*;
617
618            // size error case
619            {
620                // SAFETY: In case the error is not triggered
621                //         a segmentation fault will be triggered.
622                let too_big_slice = unsafe {
623                    //NOTE: The pointer must be initialized with a non null value
624                    //      otherwise a key constraint of slices is not fulfilled
625                    //      which can lead to crashes in release mode.
626                    use core::ptr::NonNull;
627                    core::slice::from_raw_parts(
628                        NonNull::<u8>::dangling().as_ptr(),
629                        bad_len
630                    )
631                };
632                assert_eq!(
633                    icmpv6_type.calc_checksum(ip_header.source, ip_header.destination, too_big_slice),
634                    Err(ValueTooBigError{
635                        actual: bad_len,
636                        max_allowed: (core::u32::MAX - 8) as usize,
637                        value_type: ValueType::Icmpv6PayloadLength
638                    })
639                );
640            }
641
642            // normal cases
643            {
644                let test_checksum_calc = |icmp_type: Icmpv6Type| {
645                    let expected_checksum = {
646                        crate::checksum::Sum16BitWords::new()
647                        .add_16bytes(ip_header.source)
648                        .add_16bytes(ip_header.destination)
649                        .add_2bytes([0, ip_number::IPV6_ICMP.0])
650                        .add_4bytes((
651                            payload.len() as u32 + icmpv6_type.header_len() as u32
652                        ).to_be_bytes())
653                        .add_slice(&Icmpv6Header {
654                            icmp_type: icmp_type.clone(),
655                            checksum: 0 // use zero so the checksum gets correct calculated
656                        }.to_bytes())
657                        .add_slice(&payload)
658                        .ones_complement()
659                        .to_be()
660                    };
661                    assert_eq!(
662                        expected_checksum,
663                        icmp_type.calc_checksum(
664                            ip_header.source,
665                            ip_header.destination,
666                            &payload
667                        ).unwrap()
668                    );
669                };
670
671                // unknown
672                test_checksum_calc(
673                    Unknown{
674                        type_u8, code_u8, bytes5to8
675                    }
676                );
677
678                // destination unreachable
679                for (code, _) in dest_unreachable_code_test_consts::VALID_VALUES {
680                    test_checksum_calc(DestinationUnreachable(code));
681                }
682
683                // packet too big
684                test_checksum_calc(PacketTooBig{
685                    mtu: u32::from_be_bytes(bytes5to8)
686                });
687
688                // time exceeded
689                for (code, _) in time_exceeded_code_test_consts::VALID_VALUES {
690                    test_checksum_calc(TimeExceeded(code));
691                }
692
693                // parameter problem
694                for (code, _) in parameter_problem_code_test_consts::VALID_VALUES {
695                    test_checksum_calc(ParameterProblem(
696                        ParameterProblemHeader{
697                            code,
698                            pointer: u32::from_be_bytes(bytes5to8)
699                        }
700                    ));
701                }
702
703                // echo request
704                test_checksum_calc(EchoRequest(
705                    IcmpEchoHeader::from_bytes(bytes5to8)
706                ));
707
708                // echo reply
709                test_checksum_calc(EchoReply(
710                    IcmpEchoHeader::from_bytes(bytes5to8)
711                ));
712            }
713        }

4/// Slice containing the IPv6 extension headers present after the ip header.
5///
6/// Currently supported:
7/// * Authentication Header
8/// * Hop by Hop Options Header
9/// * Destination Options Header (before and after routing headers)
10/// * Routing Header
11/// * Fragment
12/// * Authentication Header
13///
14/// Currently not supported:
15/// * Encapsulating Security Payload Header (ESP)
16/// * Host Identity Protocol (HIP)
17/// * IP Mobility
18/// * Site Multihoming by IPv6 Intermediation (SHIM6)
19#[derive(Clone, Debug, Eq, PartialEq, Default)]
20pub struct Ipv6ExtensionsSlice<'a> {
21    /// IP protocol number of the first header present in the slice.
22    first_header: Option<IpNumber>,
23    /// True if a fragment header is present in the ipv6 header extensions that causes the payload to be fragmented.
24    fragmented: bool,
25    /// Slice containing ipv6 extension headers.
26    slice: &'a [u8],
27}
28
29impl<'a> Ipv6ExtensionsSlice<'a> {
30    /// Collects all ipv6 extension headers in a slice & checks if
31    /// a fragmentation header that fragments the packet is present.
32    pub fn from_slice(
33        start_ip_number: IpNumber,
34        start_slice: &'a [u8],
35    ) -> Result<(Ipv6ExtensionsSlice, IpNumber, &'a [u8]), err::ipv6_exts::HeaderSliceError> {
36        let mut rest = start_slice;
37        let mut next_header = start_ip_number;
38        let mut fragmented = false;
39
40        use err::ipv6_exts::{HeaderError::*, HeaderSliceError::*};
41        use ip_number::*;
42
43        // the hop by hop header is required to occur directly after the ipv6 header
44        if IPV6_HOP_BY_HOP == next_header {
45            let slice = Ipv6RawExtHeaderSlice::from_slice(rest).map_err(Len)?;
46            rest = &rest[slice.slice().len()..];
47            next_header = slice.next_header();
48        }
49
50        loop {
51            match next_header {
52                IPV6_HOP_BY_HOP => {
53                    return Err(Content(HopByHopNotAtStart));
54                }
55                IPV6_DEST_OPTIONS | IPV6_ROUTE => {
56                    let slice = Ipv6RawExtHeaderSlice::from_slice(rest)
57                        .map_err(|err| Len(err.add_offset(start_slice.len() - rest.len())))?;
58                    // SAFETY:
59                    // Ipv6RawExtHeaderSlice::from_slice always generates
60                    // a subslice from the given slice rest. Therefor it is guaranteed
61                    // that len is always greater or equal the len of rest.
62                    rest = unsafe {
63                        let len = slice.slice().len();
64                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
65                    };
66                    next_header = slice.next_header();
67                }
68                IPV6_FRAG => {
69                    let slice = Ipv6FragmentHeaderSlice::from_slice(rest)
70                        .map_err(|err| Len(err.add_offset(start_slice.len() - rest.len())))?;
71                    // SAFETY:
72                    // Ipv6FragmentHeaderSlice::from_slice always generates
73                    // a subslice from the given slice rest. Therefor it is guaranteed
74                    // that len is always greater or equal the len of rest.
75                    rest = unsafe {
76                        let len = slice.slice().len();
77                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
78                    };
79                    next_header = slice.next_header();
80
81                    // check if the fragment header actually causes fragmentation
82                    fragmented = fragmented || slice.is_fragmenting_payload();
83                }
84                AUTH => {
85                    let slice = IpAuthHeaderSlice::from_slice(rest).map_err(|err| {
86                        use err::ip_auth::HeaderSliceError as I;
87                        match err {
88                            I::Len(err) => Len(err.add_offset(start_slice.len() - rest.len())),
89                            I::Content(err) => Content(IpAuth(err)),
90                        }
91                    })?;
92                    // SAFETY:
93                    // IpAuthHeaderSlice::from_slice always generates
94                    // a subslice from the given slice rest. Therefor it is guaranteed
95                    // that len is always greater or equal the len of rest.
96                    rest = unsafe {
97                        let len = slice.slice().len();
98                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
99                    };
100                    next_header = slice.next_header();
101                }
102                // done parsing, the next header is not a known/supported header extension
103                _ => break,
104            }
105        }
106
107        Ok((
108            Ipv6ExtensionsSlice {
109                first_header: if rest.len() != start_slice.len() {
110                    Some(start_ip_number)
111                } else {
112                    None
113                },
114                fragmented,
115                slice: &start_slice[..start_slice.len() - rest.len()],
116            },
117            next_header,
118            rest,
119        ))
120    }
121
122    /// Collects all ipv6 extension headers in a slice until an error
123    /// is encountered or a "non IP extension header" is found and
124    /// returns the successfully parsed parts (+ the unparsed slice
125    /// it's `IpNumber` and the error if one occurred).
126    ///
127    /// The returned values are
128    ///
129    /// * [`Ipv6ExtensionsSlice`] containing the successfully parsed IPv6 extension headers
130    /// * [`IpNumber`] of unparsed data
131    /// * Slice with unparsed data
132    /// * Optional with error if there was an error wich stoped the parsing.
133    pub fn from_slice_lax(
134        start_ip_number: IpNumber,
135        start_slice: &'a [u8],
136    ) -> (
137        Ipv6ExtensionsSlice,
138        IpNumber,
139        &'a [u8],
140        Option<(err::ipv6_exts::HeaderSliceError, err::Layer)>,
141    ) {
142        let mut rest = start_slice;
143        let mut next_header = start_ip_number;
144        let mut error = None;
145        let mut fragmented = false;
146
147        use err::ipv6_exts::{HeaderError::*, HeaderSliceError::*};
148        use ip_number::*;
149
150        // the hop by hop header is required to occur directly after the ipv6 header
151        if IPV6_HOP_BY_HOP == next_header {
152            match Ipv6RawExtHeaderSlice::from_slice(rest) {
153                Ok(slice) => {
154                    rest = &rest[slice.slice().len()..];
155                    next_header = slice.next_header();
156                }
157                Err(err) => {
158                    error = Some((Len(err), err::Layer::Ipv6HopByHopHeader));
159                }
160            }
161        }
162
163        while error.is_none() {
164            match next_header {
165                IPV6_HOP_BY_HOP => {
166                    error = Some((Content(HopByHopNotAtStart), err::Layer::Ipv6HopByHopHeader));
167                    break;
168                }
169                IPV6_DEST_OPTIONS | IPV6_ROUTE => {
170                    let slice = match Ipv6RawExtHeaderSlice::from_slice(rest) {
171                        Ok(s) => s,
172                        Err(err) => {
173                            error = Some((
174                                Len(err.add_offset(start_slice.len() - rest.len())),
175                                if next_header == IPV6_DEST_OPTIONS {
176                                    err::Layer::Ipv6DestOptionsHeader
177                                } else {
178                                    err::Layer::Ipv6RouteHeader
179                                },
180                            ));
181                            break;
182                        }
183                    };
184                    // SAFETY:
185                    // Ipv6RawExtHeaderSlice::from_slice always generates
186                    // a subslice from the given slice rest. Therefor it is guranteed
187                    // that len is always greater or equal the len of rest.
188                    rest = unsafe {
189                        let len = slice.slice().len();
190                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
191                    };
192                    next_header = slice.next_header();
193                }
194                IPV6_FRAG => {
195                    let slice = match Ipv6FragmentHeaderSlice::from_slice(rest) {
196                        Ok(s) => s,
197                        Err(err) => {
198                            error = Some((
199                                Len(err.add_offset(start_slice.len() - rest.len())),
200                                err::Layer::Ipv6FragHeader,
201                            ));
202                            break;
203                        }
204                    };
205
206                    // SAFETY:
207                    // Ipv6FragmentHeaderSlice::from_slice always generates
208                    // a subslice from the given slice rest. Therefor it is guranteed
209                    // that len is always greater or equal the len of rest.
210                    rest = unsafe {
211                        let len = slice.slice().len();
212                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
213                    };
214                    next_header = slice.next_header();
215
216                    // check if the fragment header actually causes fragmentation
217                    fragmented = fragmented || slice.is_fragmenting_payload();
218                }
219                AUTH => {
220                    use err::ip_auth::HeaderSliceError as I;
221                    let slice = match IpAuthHeaderSlice::from_slice(rest) {
222                        Ok(s) => s,
223                        Err(err) => {
224                            error = Some((
225                                match err {
226                                    I::Len(err) => {
227                                        Len(err.add_offset(start_slice.len() - rest.len()))
228                                    }
229                                    I::Content(err) => Content(IpAuth(err)),
230                                },
231                                err::Layer::IpAuthHeader,
232                            ));
233                            break;
234                        }
235                    };
236                    // SAFETY:
237                    // IpAuthHeaderSlice::from_slice always generates
238                    // a subslice from the given slice rest. Therefor it is guranteed
239                    // that len is always greater or equal the len of rest.
240                    rest = unsafe {
241                        let len = slice.slice().len();
242                        from_raw_parts(rest.as_ptr().add(len), rest.len() - len)
243                    };
244                    next_header = slice.next_header();
245                }
246                // done parsing, the next header is not a known/supported header extension
247                _ => break,
248            }
249        }
250
251        (
252            Ipv6ExtensionsSlice {
253                first_header: if rest.len() != start_slice.len() {
254                    Some(start_ip_number)
255                } else {
256                    None
257                },
258                fragmented,
259                slice: &start_slice[..start_slice.len() - rest.len()],
260            },
261            next_header,
262            rest,
263            error,
264        )
265    }
266
267    /// Returns true if a fragmentation header is present in
268    /// the extensions that fragments the payload.
269    ///
270    /// Note: A fragmentation header can still be present
271    /// even if the return value is false in case the fragmentation
272    /// headers don't fragment the payload. This is the case if
273    /// the offset of all fragmentation header is 0 and the
274    /// more fragment bit is not set.
275    #[inline]
276    pub fn is_fragmenting_payload(&self) -> bool {
277        self.fragmented
278    }
279
280    /// Returns the ip protocol number of the first header in the slice
281    /// if the slice contains an ipv6 extension header. If no ipv6 header
282    /// is present None is returned.
283    ///
284    /// None is only returned if the slice length of this struct is 0.
285    #[inline]
286    pub fn first_header(&self) -> Option<IpNumber> {
287        self.first_header
288    }
289
290    /// Slice containing the ipv6 extension headers.
291    #[inline]
292    pub fn slice(&self) -> &'a [u8] {
293        self.slice
294    }
295
296    /// Returns true if no IPv6 extension header is present (slice is empty).
297    #[inline]
298    pub fn is_empty(&self) -> bool {
299        self.slice.is_empty()
300    }
301}
302
303impl<'a> IntoIterator for Ipv6ExtensionsSlice<'a> {
304    type Item = Ipv6ExtensionSlice<'a>;
305    type IntoIter = Ipv6ExtensionSliceIter<'a>;
306
307    fn into_iter(self) -> Self::IntoIter {
308        Ipv6ExtensionSliceIter {
309            // map the next header None value to some non ipv6 ext header
310            // value.
311            next_header: self.first_header.unwrap_or(ip_number::UDP),
312            rest: self.slice,
313        }
314    }

537    proptest! {
538        #[test]
539        fn is_fragmenting_payload(
540            hop_by_hop_options in ipv6_raw_ext_any(),
541            destination_options in ipv6_raw_ext_any(),
542            routing in ipv6_raw_ext_any(),
543            auth in ip_auth_any(),
544            final_destination_options in ipv6_raw_ext_any()
545        ) {
546            // no fragment header
547            {
548                let mut exts = Ipv6Extensions{
549                    hop_by_hop_options: Some(hop_by_hop_options),
550                    destination_options: Some(destination_options),
551                    routing: Some(
552                        Ipv6RoutingExtensions {
553                            routing,
554                            final_destination_options: Some(final_destination_options),
555                        }
556                    ),
557                    fragment: None,
558                    auth: Some(auth),
559                };
560                let first_ip_number = exts.set_next_headers(UDP);
561
562                let mut bytes = Vec::with_capacity(exts.header_len());
563                exts.write(&mut bytes, first_ip_number).unwrap();
564
565                let (header, _, _) = Ipv6ExtensionsSlice::from_slice(first_ip_number, &bytes).unwrap();
566                assert_eq!(false, header.is_fragmenting_payload());
567            }
568
569            // different variants of the fragment header with
570            // variants that fragment and variants that don't fragment
571            let frag_variants : [(bool, Ipv6FragmentHeader);4] = [
572                (false, Ipv6FragmentHeader::new(UDP, 0.try_into().unwrap(), false, 123)),
573                (true, Ipv6FragmentHeader::new(UDP, 2.try_into().unwrap(), false, 123)),
574                (true, Ipv6FragmentHeader::new(UDP, 0.try_into().unwrap(), true, 123)),
575                (true, Ipv6FragmentHeader::new(UDP, 3.try_into().unwrap(), true, 123)),
576            ];
577
578            for (first_expected, first_header) in frag_variants.iter() {
579                // single fragment header
580                {
581                    let bytes = first_header.to_bytes();
582                    let (header, _, _) = Ipv6ExtensionsSlice::from_slice(IPV6_FRAG, &bytes).unwrap();
583                    assert_eq!(*first_expected, header.is_fragmenting_payload());
584                }
585                // two fragment headers
586                for (second_expected, second_header) in frag_variants.iter() {
587                    let mut first_mod = first_header.clone();
588                    first_mod.next_header = IPV6_FRAG;
589                    let mut bytes = Vec::with_capacity(first_mod.header_len() + second_header.header_len());
590                    bytes.extend_from_slice(&first_mod.to_bytes());
591                    bytes.extend_from_slice(&second_header.to_bytes());
592
593                    let (header, _, _) = Ipv6ExtensionsSlice::from_slice(IPV6_FRAG, &bytes).unwrap();
594                    assert_eq!(
595                        *first_expected || *second_expected,
596                        header.is_fragmenting_payload()
597                    );
598                }
599            }
600        }
601    }

603    #[test]
604    fn is_empty() {
605        // empty
606        {
607            let slice = Ipv6ExtensionsSlice::from_slice(ip_number::UDP, &[])
608                .unwrap()
609                .0;
610            assert!(slice.is_empty());
611        }
612
613        // fragment
614        {
615            let bytes =
616                Ipv6FragmentHeader::new(ip_number::UDP, IpFragOffset::ZERO, true, 0).to_bytes();
617            let slice = Ipv6ExtensionsSlice::from_slice(ip_number::IPV6_FRAG, &bytes)
618                .unwrap()
619                .0;
620            assert_eq!(false, slice.is_empty());
621        }

41pub fn analyze_packet(data: &[u8]) -> Result<PacketAnalysis> {
42    if data.len() < 14 {
43        return Err(anyhow!("Packet too small to be valid"));
44    }
45
46    let mut analysis = PacketAnalysis {
47        packet_type: "ethernet".to_string(),
48        source_ip: None,
49        dest_ip: None,
50        source_port: None,
51        dest_port: None,
52        protocol: "unknown".to_string(),
53        payload_size: 0,
54        flags: Vec::new(),
55        timestamp: Some(chrono::Utc::now().timestamp()),
56    };
57
58    match SlicedPacket::from_ethernet(data) {
59        Ok(packet) => {
60            // Process IP layer with simplified approach
61            if let Some(net) = &packet.net {
62                match net {
63                    NetSlice::Ipv4(ipv4_slice) => {
64                        // Extract source and destination using header methods
65                        let header = ipv4_slice.header();
66                        analysis.source_ip = Some(Ipv4Addr::from(header.source()).to_string());
67                        analysis.dest_ip = Some(Ipv4Addr::from(header.destination()).to_string());
68                        analysis.protocol = protocol_name(header.protocol()).to_string();
69                    }
70                    NetSlice::Ipv6(ipv6_slice) => {
71                        // Extract source and destination using header methods
72                        let header = ipv6_slice.header();
73                        analysis.source_ip = Some(header.source().to_string());
74                        analysis.dest_ip = Some(header.destination().to_string());
75                        analysis.protocol = protocol_name(header.next_header()).to_string();
76                    }
77                }
78            }
79
80            // Process transport layer
81            if let Some(transport) = &packet.transport {
82                match transport {
83                    TransportSlice::Tcp(tcp_slice) => {
84                        analysis.protocol = "TCP".to_string();
85                        // Extract ports using header methods
86                        analysis.source_port = Some(tcp_slice.source_port());
87                        analysis.dest_port = Some(tcp_slice.destination_port());
88                        
89                        // Extract flags from header
90                        let header = tcp_slice.to_header();
91                        let mut flags = Vec::new();
92                        if header.fin { flags.push("FIN".to_string()); }
93                        if header.syn { flags.push("SYN".to_string()); }
94                        if header.rst { flags.push("RST".to_string()); }
95                        if header.psh { flags.push("PSH".to_string()); }
96                        if header.ack { flags.push("ACK".to_string()); }
97                        if header.urg { flags.push("URG".to_string()); }
98                        if header.ece { flags.push("ECE".to_string()); }
99                        if header.cwr { flags.push("CWR".to_string()); }
100                        analysis.flags = flags;
101                    }
102                    TransportSlice::Udp(udp_slice) => {
103                        analysis.protocol = "UDP".to_string();
104                        // Extract ports using header methods
105                        analysis.source_port = Some(udp_slice.source_port());
106                        analysis.dest_port = Some(udp_slice.destination_port());
107                    }
108                    TransportSlice::Icmpv4(_) => {
109                        analysis.protocol = "ICMP".to_string();
110                    }
111                    TransportSlice::Icmpv6(_) => {
112                        analysis.protocol = "ICMPv6".to_string();
113                    }
114                }
115            }
116
117            // Calculate payload size
118            let header_size = 14 + // Ethernet
119                packet.net.as_ref().map(|n| match n {
120                    NetSlice::Ipv4(s) => s.header_len() as usize,
121                    NetSlice::Ipv6(s) => 40, // IPv6 header is always 40 bytes
122                }).unwrap_or(0) +
123                packet.transport.as_ref().map(|t| match t {
124                    TransportSlice::Tcp(s) => s.data_offset() as usize * 4, // TCP data offset is in 32-bit words
125                    TransportSlice::Udp(s) => 8, // UDP header is always 8 bytes
126                    TransportSlice::Icmpv4(_) => 8, // ICMP header is typically 8 bytes
127                    TransportSlice::Icmpv6(_) => 8, // ICMPv6 header is typically 8 bytes
128                }).unwrap_or(0);
129            
130            analysis.payload_size = data.len().saturating_sub(header_size);
131
132            Ok(analysis)
133        }
134        Err(e) => Err(anyhow!("Failed to parse packet: {}", e)),
135    }
136}

138pub fn parse_packet_details(data: &[u8]) -> Result<PacketInfo> {
139    let packet = SlicedPacket::from_ethernet(data)
140        .map_err(|e| anyhow!("Failed to parse packet: {}", e))?;
141
142    let ethernet = if let Some(eth) = &packet.link {
143        match eth {
144            LinkSlice::Ethernet2(eth_slice) => {
145                let header = eth_slice.to_header();
146                Some(EthernetInfo {
147                    source_mac: format_mac(&header.source),
148                    destination_mac: format_mac(&header.destination),
149                    ether_type: format!("{:04x}", header.ether_type),
150                })
151            }
152            _ => None,
153        }
154    } else {
155        None
156    };
157
158    let ip = if let Some(net_slice) = &packet.net {
159        match net_slice {
160            NetSlice::Ipv4(ipv4_slice) => {
161                let header = ipv4_slice.to_header();
162                Some(IpInfo {
163                    version: 4,
164                    source: Ipv4Addr::from(header.source).to_string(),
165                    destination: Ipv4Addr::from(header.destination).to_string(),
166                    protocol: protocol_name(header.protocol).to_string(),
167                    ttl: Some(header.time_to_live),
168                    total_length: Some(header.total_len),
169                })
170            }
171            NetSlice::Ipv6(ipv6_slice) => {
172                let header = ipv6_slice.to_header();
173                Some(IpInfo {
174                    version: 6,
175                    source: header.source.to_string(),
176                    destination: header.destination.to_string(),
177                    protocol: protocol_name(header.next_header).to_string(),
178                    ttl: Some(header.hop_limit),
179                    total_length: Some(header.payload_length + 40), // IPv6 header is 40 bytes
180                })
181            }
182        }
183    } else {
184        None
185    };
186
187    let transport = if let Some(trans) = &packet.transport {
188        match trans {
189            TransportSlice::Tcp(tcp_slice) => {
190                let header = tcp_slice.to_header();
191                let mut flags = Vec::new();
192                if header.fin { flags.push("FIN".to_string()); }
193                if header.syn { flags.push("SYN".to_string()); }
194                if header.rst { flags.push("RST".to_string()); }
195                if header.psh { flags.push("PSH".to_string()); }
196                if header.ack { flags.push("ACK".to_string()); }
197                if header.urg { flags.push("URG".to_string()); }
198                if header.ece { flags.push("ECE".to_string()); }
199                if header.cwr { flags.push("CWR".to_string()); }
200                
201                Some(TransportInfo {
202                    protocol: "TCP".to_string(),
203                    source_port: Some(tcp_slice.source_port()),
204                    destination_port: Some(tcp_slice.destination_port()),
205                    flags,
206                })
207            }
208            TransportSlice::Udp(udp_slice) => {
209                Some(TransportInfo {
210                    protocol: "UDP".to_string(),
211                    source_port: Some(udp_slice.source_port()),
212                    destination_port: Some(udp_slice.destination_port()),
213                    flags: vec![],
214                })
215            }
216            _ => None,
217        }
218    } else {
219        None
220    };
221
222    // Calculate sizes
223    let header_size = 14 + // Ethernet
224        packet.net.as_ref().map(|n| match n {
225            NetSlice::Ipv4(s) => s.header_len() as usize,
226            NetSlice::Ipv6(s) => 40, // IPv6 header is always 40 bytes
227        }).unwrap_or(0) +
228        packet.transport.as_ref().map(|t| match t {
229            TransportSlice::Tcp(s) => s.data_offset() as usize * 4, // TCP data offset is in 32-bit words
230            TransportSlice::Udp(s) => 8, // UDP header is always 8 bytes
231            TransportSlice::Icmpv4(_) => 8, // ICMP header is typically 8 bytes
232            TransportSlice::Icmpv6(_) => 8, // ICMPv6 header is typically 8 bytes
233        }).unwrap_or(0);
234
235    Ok(PacketInfo {
236        ethernet,
237        ip,
238        transport,
239        payload_size: data.len().saturating_sub(header_size),
240        total_size: data.len(),
241    })
242}

244fn format_mac(mac: &[u8; 6]) -> String {
245    format!(
246        "{:02x}:{:02x}:{:02x}:{:02x}:{:02x}:{:02x}",
247        mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]
248    )
249}

251fn protocol_name(proto: u8) -> &'static str {
252    match proto {
253        1 => "ICMP",
254        6 => "TCP",
255        17 => "UDP",
256        41 => "IPv6",
257        47 => "GRE",
258        50 => "ESP",
259        51 => "AH",
260        58 => "ICMPv6",
261        _ => "Unknown",
262    }
263}

265fn get_tcp_flags_from_byte(flags_byte: u8) -> Vec<String> {
266    let mut flags = Vec::new();
267    
268    if flags_byte & 0x01 != 0 { flags.push("FIN".to_string()); }
269    if flags_byte & 0x02 != 0 { flags.push("SYN".to_string()); }
270    if flags_byte & 0x04 != 0 { flags.push("RST".to_string()); }
271    if flags_byte & 0x08 != 0 { flags.push("PSH".to_string()); }
272    if flags_byte & 0x10 != 0 { flags.push("ACK".to_string()); }
273    if flags_byte & 0x20 != 0 { flags.push("URG".to_string()); }
274    if flags_byte & 0x40 != 0 { flags.push("ECE".to_string()); }
275    if flags_byte & 0x80 != 0 { flags.push("CWR".to_string()); }
276    
277    flags
278}

280// Check for suspicious packet characteristics
281pub fn check_packet_anomalies(packet: &PacketAnalysis) -> Vec<String> {
282    let mut anomalies = Vec::new();
283
284    // Check for suspicious port combinations
285    if let (Some(sport), Some(dport)) = (packet.source_port, packet.dest_port) {
286        // Common malware ports
287        let suspicious_ports = vec![135, 139, 445, 1433, 3389, 4444, 5555, 6666, 7777, 8888, 9999];
288        if suspicious_ports.contains(&sport) || suspicious_ports.contains(&dport) {
289            anomalies.push("Suspicious port detected".to_string());
290        }
291
292        // Check for port 0 (invalid)
293        if sport == 0 || dport == 0 {
294            anomalies.push("Invalid port 0 detected".to_string());
295        }
296    }
297
298    // Check TCP flags anomalies
299    if packet.protocol == "TCP" && !packet.flags.is_empty() {
300        // SYN+FIN is invalid
301        if packet.flags.contains(&"SYN".to_string()) && packet.flags.contains(&"FIN".to_string()) {
302            anomalies.push("Invalid TCP flags: SYN+FIN".to_string());
303        }
304
305        // NULL scan (no flags)
306        if packet.flags.is_empty() {
307            anomalies.push("NULL TCP scan detected".to_string());
308        }
309
310        // XMAS scan (FIN+PSH+URG)
311        if packet.flags.contains(&"FIN".to_string()) 
312            && packet.flags.contains(&"PSH".to_string()) 
313            && packet.flags.contains(&"URG".to_string()) {
314            anomalies.push("XMAS TCP scan detected".to_string());
315        }
316    }
317
318    // Check for unusually small or large packets
319    if packet.payload_size == 0 && packet.protocol == "TCP" {
320        if !packet.flags.contains(&"SYN".to_string()) 
321            && !packet.flags.contains(&"FIN".to_string()) 
322            && !packet.flags.contains(&"RST".to_string()) {
323            anomalies.push("Empty TCP payload (possible scan)".to_string());
324        }
325    }
326
327    anomalies
328}

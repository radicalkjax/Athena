36pub fn detect_anomalies(traffic_data: &str) -> Result<Vec<NetworkAnomaly>> {
37    let packets: Vec<PacketAnalysis> = serde_json::from_str(traffic_data)
38        .map_err(|e| anyhow!("Failed to parse traffic data: {}", e))?;
39
40    let mut anomalies = Vec::new();
41
42    // Detect various types of anomalies
43    if let Some(anomaly) = detect_packet_flood(&packets) {
44        anomalies.push(anomaly);
45    }
46
47    if let Some(anomaly) = detect_protocol_anomalies(&packets) {
48        anomalies.push(anomaly);
49    }
50
51    if let Some(anomaly) = detect_timing_anomalies(&packets) {
52        anomalies.push(anomaly);
53    }
54
55    if let Some(anomaly) = detect_payload_anomalies(&packets) {
56        anomalies.push(anomaly);
57    }
58
59    Ok(anomalies)
60}

62pub fn detect_port_scan(packets_json: &str) -> Result<Value> {
63    let packets: Vec<PacketAnalysis> = serde_json::from_str(packets_json)
64        .map_err(|e| anyhow!("Failed to parse packets: {}", e))?;
65
66    let mut source_activity: HashMap<String, PortScanActivity> = HashMap::new();
67
68    // Analyze packet patterns
69    for packet in &packets {
70        if let (Some(src), Some(dst), Some(port)) = (&packet.source_ip, &packet.dest_ip, packet.dest_port) {
71            let activity = source_activity.entry(src.clone()).or_insert_with(|| PortScanActivity {
72                targets: HashSet::new(),
73                ports: HashSet::new(),
74                syn_count: 0,
75                ack_count: 0,
76                rst_count: 0,
77                first_seen: packet.timestamp.unwrap_or(0),
78                last_seen: packet.timestamp.unwrap_or(0),
79            });
80
81            activity.targets.insert(dst.clone());
82            activity.ports.insert(port);
83            activity.last_seen = packet.timestamp.unwrap_or(0);
84
85            // Count TCP flags
86            if packet.flags.contains(&"SYN".to_string()) && !packet.flags.contains(&"ACK".to_string()) {
87                activity.syn_count += 1;
88            }
89            if packet.flags.contains(&"ACK".to_string()) {
90                activity.ack_count += 1;
91            }
92            if packet.flags.contains(&"RST".to_string()) {
93                activity.rst_count += 1;
94            }
95        }
96    }
97
98    // Identify port scans
99    let mut scan_infos = Vec::new();
100    
101    for (source, activity) in source_activity {
102        let scan_type = identify_scan_type(&activity);
103        
104        if !scan_type.is_empty() {
105            scan_infos.push(PortScanInfo {
106                scan_type,
107                source_ip: source,
108                target_ips: activity.targets.into_iter().collect(),
109                scanned_ports: activity.ports.into_iter().collect(),
110                duration_ms: (activity.last_seen - activity.first_seen) as u64 * 1000,
111                packet_count: activity.syn_count + activity.ack_count + activity.rst_count,
112            });
113        }
114    }
115
116    Ok(json!({
117        "scans_detected": scan_infos.len(),
118        "scan_details": scan_infos,
119    }))
120}

122pub fn detect_data_exfiltration(traffic_json: &str) -> Result<Value> {
123    let packets: Vec<PacketAnalysis> = serde_json::from_str(traffic_json)
124        .map_err(|e| anyhow!("Failed to parse traffic: {}", e))?;
125
126    let mut flow_stats: HashMap<String, FlowStatistics> = HashMap::new();
127
128    // Collect flow statistics
129    for packet in &packets {
130        if let (Some(src), Some(dst)) = (&packet.source_ip, &packet.dest_ip) {
131            let flow_key = format!("{}->{}", src, dst);
132            
133            let stats = flow_stats.entry(flow_key.clone()).or_insert_with(|| FlowStatistics {
134                source: src.clone(),
135                destination: dst.clone(),
136                total_bytes: 0,
137                packet_count: 0,
138                first_seen: packet.timestamp.unwrap_or(0),
139                last_seen: packet.timestamp.unwrap_or(0),
140                protocol: packet.protocol.clone(),
141            });
142
143            stats.total_bytes += packet.payload_size;
144            stats.packet_count += 1;
145            stats.last_seen = packet.timestamp.unwrap_or(0);
146        }
147    }
148
149    // Analyze for exfiltration patterns
150    let mut exfiltration_candidates = Vec::new();
151
152    for (_, stats) in flow_stats {
153        let risk_score = calculate_exfiltration_risk(&stats);
154        
155        if risk_score > 0.5 {
156            exfiltration_candidates.push(DataExfiltrationInfo {
157                source_ip: stats.source,
158                destination_ip: stats.destination,
159                total_bytes: stats.total_bytes,
160                duration_ms: (stats.last_seen - stats.first_seen) as u64 * 1000,
161                protocol: stats.protocol,
162                risk_score,
163            });
164        }
165    }
166
167    // Sort by risk score
168    exfiltration_candidates.sort_by(|a, b| b.risk_score.partial_cmp(&a.risk_score).unwrap());
169
170    Ok(json!({
171        "exfiltration_detected": !exfiltration_candidates.is_empty(),
172        "high_risk_flows": exfiltration_candidates.len(),
173        "flows": exfiltration_candidates,
174    }))
175}

177fn detect_packet_flood(packets: &[PacketAnalysis]) -> Option<NetworkAnomaly> {
178    let mut packet_rate: HashMap<i64, usize> = HashMap::new();
179    
180    // Count packets per second
181    for packet in packets {
182        if let Some(ts) = packet.timestamp {
183            *packet_rate.entry(ts).or_insert(0) += 1;
184        }
185    }
186
187    // Check for flood
188    let high_rate_seconds = packet_rate.values().filter(|&&rate| rate > 1000).count();
189    
190    if high_rate_seconds > 5 {
191        Some(NetworkAnomaly {
192            anomaly_type: "Packet Flood".to_string(),
193            severity: "High".to_string(),
194            description: format!("Detected {} seconds with >1000 packets/sec", high_rate_seconds),
195            indicators: vec![
196                "Excessive packet rate".to_string(),
197                "Possible DoS attack".to_string(),
198            ],
199            timestamp: chrono::Utc::now().timestamp(),
200        })
201    } else {
202        None
203    }
204}

206fn detect_protocol_anomalies(packets: &[PacketAnalysis]) -> Option<NetworkAnomaly> {
207    let mut anomaly_indicators = Vec::new();
208    
209    for packet in packets {
210        // Check for malformed TCP flags
211        if packet.protocol == "TCP" {
212            if packet.flags.contains(&"SYN".to_string()) && packet.flags.contains(&"FIN".to_string()) {
213                anomaly_indicators.push("Invalid TCP flags combination (SYN+FIN)".to_string());
214            }
215            
216            if packet.flags.is_empty() && packet.payload_size == 0 {
217                anomaly_indicators.push("NULL TCP packet detected".to_string());
218            }
219        }
220
221        // Check for unusual protocols
222        if matches!(packet.protocol.as_str(), "Unknown" | "ICMP" | "ICMPv6") {
223            if packet.payload_size > 1000 {
224                anomaly_indicators.push(format!("Large {} packet ({}B)", packet.protocol, packet.payload_size));
225            }
226        }
227    }
228
229    if !anomaly_indicators.is_empty() {
230        Some(NetworkAnomaly {
231            anomaly_type: "Protocol Anomaly".to_string(),
232            severity: "Medium".to_string(),
233            description: "Detected unusual protocol behavior".to_string(),
234            indicators: anomaly_indicators,
235            timestamp: chrono::Utc::now().timestamp(),
236        })
237    } else {
238        None
239    }
240}

242fn detect_timing_anomalies(packets: &[PacketAnalysis]) -> Option<NetworkAnomaly> {
243    if packets.len() < 10 {
244        return None;
245    }
246
247    let mut timestamps: Vec<i64> = packets
248        .iter()
249        .filter_map(|p| p.timestamp)
250        .collect();
251    
252    timestamps.sort();
253
254    // Check for burst patterns
255    let mut burst_count = 0;
256    for i in 1..timestamps.len() {
257        if timestamps[i] - timestamps[i-1] < 10 {
258            burst_count += 1;
259        }
260    }
261
262    if burst_count as f64 / timestamps.len() as f64 > 0.7 {
263        Some(NetworkAnomaly {
264            anomaly_type: "Timing Anomaly".to_string(),
265            severity: "Medium".to_string(),
266            description: "Detected burst traffic pattern".to_string(),
267            indicators: vec![
268                format!("{}% of packets in bursts", (burst_count * 100 / timestamps.len())),
269                "Possible automated traffic".to_string(),
270            ],
271            timestamp: chrono::Utc::now().timestamp(),
272        })
273    } else {
274        None
275    }
276}

278fn detect_payload_anomalies(packets: &[PacketAnalysis]) -> Option<NetworkAnomaly> {
279    let mut anomalies = Vec::new();
280    let mut total_size = 0;
281    let mut large_packets = 0;
282
283    for packet in packets {
284        total_size += packet.payload_size;
285        
286        // Check for unusually large payloads
287        if packet.payload_size > 10000 {
288            large_packets += 1;
289        }
290
291        // Check for specific patterns in small packets
292        if packet.payload_size == 1 || packet.payload_size == 0 {
293            if packet.protocol == "TCP" && !packet.flags.contains(&"FIN".to_string()) {
294                anomalies.push("Suspicious small payload".to_string());
295            }
296        }
297    }
298
299    if large_packets > 10 || !anomalies.is_empty() {
300        Some(NetworkAnomaly {
301            anomaly_type: "Payload Anomaly".to_string(),
302            severity: if large_packets > 50 { "High" } else { "Medium" }.to_string(),
303            description: "Detected unusual payload patterns".to_string(),
304            indicators: {
305                let mut indicators = anomalies;
306                if large_packets > 0 {
307                    indicators.push(format!("{} large packets detected", large_packets));
308                }
309                indicators
310            },
311            timestamp: chrono::Utc::now().timestamp(),
312        })
313    } else {
314        None
315    }
316}

339fn identify_scan_type(activity: &PortScanActivity) -> String {
340    let port_count = activity.ports.len();
341    let target_count = activity.targets.len();
342    
343    if port_count > 100 && target_count == 1 {
344        "Vertical Port Scan".to_string()
345    } else if port_count < 10 && target_count > 10 {
346        "Horizontal Port Scan".to_string()
347    } else if port_count > 20 && target_count > 5 {
348        "Block Scan".to_string()
349    } else if activity.syn_count > 0 && activity.ack_count == 0 {
350        "SYN Scan".to_string()
351    } else if activity.rst_count > activity.syn_count / 2 {
352        "Stealth Scan".to_string()
353    } else if port_count > 20 || target_count > 20 {
354        "General Scan".to_string()
355    } else {
356        String::new()
357    }
358}

360fn calculate_exfiltration_risk(stats: &FlowStatistics) -> f64 {
361    let mut risk = 0.0;
362    
363    // Large data transfer
364    if stats.total_bytes > 100_000_000 {
365        risk += 0.4;
366    } else if stats.total_bytes > 10_000_000 {
367        risk += 0.3;
368    }
369
370    // Long duration
371    let duration_hours = (stats.last_seen - stats.first_seen) as f64 / 3600.0;
372    if duration_hours > 1.0 {
373        risk += 0.2;
374    }
375
376    // High packet rate with large payloads
377    let avg_packet_size = stats.total_bytes as f64 / stats.packet_count as f64;
378    if avg_packet_size > 1000.0 && stats.packet_count > 1000 {
379        risk += 0.3;
380    }
381
382    // External destination (simplified check)
383    if !stats.destination.starts_with("192.168.") 
384        && !stats.destination.starts_with("10.") 
385        && !stats.destination.starts_with("172.") {
386        risk += 0.2;
387    }
388
389    risk.min(1.0)
390}
